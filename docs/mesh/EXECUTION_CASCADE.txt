message = 'Secret: Meet at midnight'

encrypted = enc_a.encrypt_message(message, enc_b.public_key)

print(f'  {message}')

STEP 7: SHOW MULTI-HOP ROUTING (5 minutes)
════════════════════════════════════════════════════════════════════

Demonstrate message path finding

Command:
  python -c "
import sys
import asyncio
sys.path.insert(0, 'backend/core')
from routing import MeshRouter

async def show_routing():
    router_a = MeshRouter('A')
    
    # Create A - B - C network
    await router_a.add_link('A', 'B', -50, 5)
    await router_a.add_link('B', 'C', -55, 5)
    
    # Find paths
    path_to_b = router_a.get_path('A', 'B')
    path_to_c = router_a.get_path('A', 'C')
    
    print('MESH NETWORK TOPOLOGY:')
    print('  A ---[50m]--- B ---[60m]--- C')
    print()
    print('ROUTING PATHS FROM A:')
    print(f'  To B: {\" → \".join(path_to_b)} ({len(path_to_b)-1} hop)')
    print(f'  To C: {\" → \".join(path_to_c)} ({len(path_to_c)-1} hops)')
    print()
    print('✅ OLSR Routing working!')

asyncio.run(show_routing())
"

Expected Output:
  MESH NETWORK TOPOLOGY:
    A ---[50m]--- B ---[60m]--- C

  ROUTING PATHS FROM A:
    To B: A → B (1 hop)
    To C: A → B → C (2 hops)

  ✅ OLSR Routing working!
